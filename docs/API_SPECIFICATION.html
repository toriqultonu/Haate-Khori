1. Project Overview

HaateKhori is an educational Android app for kids featuring multiple learning games. The backend needs to provide content data for the following modules:

  • Alphabet Learning: 26 letters with words, sentences, and emojis
  • Word Search: Multiple topics with word lists for puzzle generation
  • Counting Game: Categories with emoji items for counting practice
  • Memory Match: Card pair categories for memory game
  • Pattern Recognition: "What comes next?" pattern sequences
  • Stick Builder: 7-segment display number building game
  • Coloring Book: Coloring pages with shape definitions and color palette

4.3 PatternSequence Entity

@Entity
@Table(name = "pattern_sequences")
data class PatternSequence(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,

    @Column(name = "pattern_id", nullable = false, unique = true, length = 50)
    val patternId: String,

    @Column(nullable = false, length = 50)
    val category: String,

    @Column(nullable = false, columnDefinition = "JSON")
    @Convert(converter = StringListConverter::class)
    val sequence: List<String>,

    @Column(nullable = false, length = 50)
    val answer: String,

    @Column(nullable = false, columnDefinition = "JSON")
    @Convert(converter = StringListConverter::class)
    val options: List<String>,

    @Column(nullable = false)
    val difficulty: Int = 1,

    @Column(name = "is_active", nullable = false)
    val isActive: Boolean = true
)

// JSON Converter
@Converter
class StringListConverter : AttributeConverter<List<String>, String> {
    private val objectMapper = ObjectMapper()

    override fun convertToDatabaseColumn(attribute: List<String>?): String {
        return objectMapper.writeValueAsString(attribute ?: emptyList<String>())
    }

    override fun convertToEntityAttribute(dbData: String?): List<String> {
        return dbData?.let {
            objectMapper.readValue(it, object : TypeReference<List<String>>() {})
        } ?: emptyList()
    }
}

7.3 Global Exception Handler

@RestControllerAdvice
class GlobalExceptionHandler {

    @ExceptionHandler(ResourceNotFoundException::class)
    fun handleNotFound(ex: ResourceNotFoundException): ResponseEntity<ErrorResponse> {
        val error = ErrorResponse(
            success = false,
            error = ErrorDetails(
                code = "RESOURCE_NOT_FOUND",
                message = ex.message ?: "Resource not found",
                details = null,
                path = null
            ),
            timestamp = LocalDateTime.now()
        )
        return ResponseEntity.status(HttpStatus.NOT_FOUND).body(error)
    }

    @ExceptionHandler(MethodArgumentNotValidException::class)
    fun handleValidation(ex: MethodArgumentNotValidException): ResponseEntity<ErrorResponse> {
        val details = ex.bindingResult.fieldErrors
            .map { "${it.field}: ${it.defaultMessage}" }
            .joinToString(", ")

        val error = ErrorResponse(
            success = false,
            error = ErrorDetails(
                code = "VALIDATION_ERROR",
                message = "Validation failed",
                details = details,
                path = null
            ),
            timestamp = LocalDateTime.now()
        )
        return ResponseEntity.status(HttpStatus.BAD_REQUEST).body(error)
    }

    @ExceptionHandler(Exception::class)
    fun handleGeneral(ex: Exception): ResponseEntity<ErrorResponse> {
        val error = ErrorResponse(
            success = false,
            error = ErrorDetails(
                code = "INTERNAL_ERROR",
                message = "An unexpected error occurred",
                details = null,
                path = null
            ),
            timestamp = LocalDateTime.now()
        )
        return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).body(error)
    }
}

data class ErrorResponse(

    val error: ErrorDetails,
    val timestamp: LocalDateTime
)

data class ErrorDetails(
    val code: String,
    val message: String,
    val details: String?,
    val path: String?
)
